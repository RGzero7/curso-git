-Limpiar pantalla
clear

-inicializar git en la carpeta
git init

-tomar en cuenta un archivo
git add nombre.extencion

-agregar todos los archivos
git add .

-dejar de tomar en cuenta archivo
git rm --cached nombre.extencion

-etiquetado "commit"
git commit -m "mensaje"

-pasos para agregar a github
echo "# curso-git" >> README.md
git init
git add README.md
git commit -m "first commit"
git branch -M main
git remote add origin https://github.com/RGzero7/curso-git.git
git push -u origin main

-saber en que rama estoy
git branch

-ver historial de commit realizados
git log
 q para terminar o enter para ver cada commit hasta el final

- ver dirección de GitHub donde se guardan los commit
git remote -v

- ver los cambios realizados
git diff

-descartar el ultimo add realizado
git restore --staged .

-descartar cambios
git restore .\nombre.extencion
git restore . (restaura todo)

el proceso para descartar lo últimos cambios es
git restore --staged .
git restore .

-devolver todo a un commit en especifico
git log
copiamos id del commit 
git reset --hard aquielID

- Ramas -
- ver ramas y la activa actualmente
git branch
-agregar nueva rama
git branch nombre
-moverme entre ramas
git checkout nombre
-agregar y moverme a la nueva rama
git checkout -b nombre

-juntar ramas
primero volver a main
git checkout main
git merge nombre (de la rama que queremos volcar en main)
para terminar debe realizarse un add y un commit

-abortar junta
git merge --abort

-eliminar ramas
git branch -d nombre (si hay conflictos como no haber realizado merge, cambia la d minúscula por una mayuscula)

-actualizar el repositorio local con el guardado en GitHub
git pull origin main

-ahora del local actualizar el repositorio en GitHub
git push origin main
